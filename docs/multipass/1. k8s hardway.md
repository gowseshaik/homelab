#!/bin/bash
# Step 1: Create Multipass VMs

# Master Nodes
multipass launch -n k8s-master-1 --cpus 2 --mem 2G --disk 10G
multipass launch -n k8s-master-2 --cpus 2 --mem 2G --disk 10G

# Worker Nodes
multipass launch -n k8s-worker-1 --cpus 2 --mem 2G --disk 10G
multipass launch -n k8s-worker-2 --cpus 2 --mem 2G --disk 10G

# Load Balancer
multipass launch -n k8s-lb --cpus 1 --mem 1G --disk 5G

# Step 2: Get IPs
MASTER1_IP=$(multipass info k8s-master-1 | grep IPv4 | awk '{print $2}')
MASTER2_IP=$(multipass info k8s-master-2 | grep IPv4 | awk '{print $2}')
WORKER1_IP=$(multipass info k8s-worker-1 | grep IPv4 | awk '{print $2}')
WORKER2_IP=$(multipass info k8s-worker-2 | grep IPv4 | awk '{print $2}')
LB_IP=$(multipass info k8s-lb | grep IPv4 | awk '{print $2}')

# Step 3: Install Prerequisites on all nodes (Docker, curl, etc.)
for NODE in k8s-master-1 k8s-master-2 k8s-worker-1 k8s-worker-2 k8s-lb; do
  multipass exec $NODE -- bash -c '
    sudo apt-get update && \
    sudo apt-get install -y docker.io curl wget vim
  '
done

# Step 4: Download Kubernetes Binaries (cfssl, kubectl, etc.) on host

# Step 6: Configure etcd cluster on both master nodes

# Step 7: Configure Kubernetes control plane static pods
# Create manifests for kube-apiserver, controller-manager, scheduler
# Set --advertise-address, --bind-address, --etcd-servers, --secure-port

# Step 8: Configure kubelet, kube-proxy on worker nodes

# Step 9: Setup Load Balancer using HAProxy or nginx
# On k8s-lb node, install HAProxy and configure to forward port 6443 to both masters
# Example:

#   bind *:6443
#   default_backend k8s-masters
# backend k8s-masters
#   balance roundrobin
#   server master1 MASTER1_IP:6443 check
